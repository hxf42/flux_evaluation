import matplotlib.pyplot as plt
plt.xlabel('Fruit')
fruits = ['Apple', 'Banana', 'Watermelon', 'Lemon', 'Orange']
plt.ylabel('Number')
numbers = [10, 24, 36, 48, 20]
plt.bar(fruits, numbers)
plt.title('Number of Fruits')
plt.show()
